cmake_minimum_required(VERSION 3.21)

# Commercial white-label build. When ON, the "Powered by SHD Systems" attribution
# is removed from the setup window. This is only licensed for use by holders of a
# commercial licence (see COMMERCIAL-LICENSE.md); under the free AGPLv3 licence the
# attribution must remain. Default OFF.
option(SHD_WHITELABEL "Remove SHD attribution (requires a commercial licence)" OFF)

# Generic, config-driven Qt installer GUI. All product-specific data lives in
# resources/config.json (generated by pack.ps1), so this target itself never
# needs editing between projects — it always builds "AppSetup.exe".
project(app-setup LANGUAGES CXX)

set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_AUTOMOC ON)
set(CMAKE_AUTORCC ON)

find_package(Qt6 REQUIRED COMPONENTS Widgets Svg)

set(RES "${CMAKE_CURRENT_SOURCE_DIR}/resources")

# config.json must exist for AUTORCC. pack.ps1 writes it; provide a stub for a
# bare `cmake --build` so the tree at least configures.
if(NOT EXISTS "${RES}/config.json")
    file(WRITE "${RES}/config.json"
         "{\n  \"appName\": \"App\",\n  \"registryKey\": \"App\",\n  \"version\": \"0.0.0\",\n  \"apps\": []\n}\n")
endif()
# app.ico is binary; if the project hasn't supplied one yet, AUTORCC/windres will
# fail. pack.ps1 copies the real icon in before building.
if(NOT EXISTS "${RES}/app.ico")
    message(WARNING "resources/app.ico is missing — run pack.ps1 or add one before building.")
endif()

set(SETUP_SOURCES
    src/main.cpp
    src/setupwindow.cpp
    src/setupwindow.h
    ${RES}/setup.qrc
)

if(WIN32)
    # Copy the manifest/icon next to the generated .rc and reference them by bare
    # filename so windres never sees a path with spaces.
    configure_file(${RES}/setup.manifest ${CMAKE_CURRENT_BINARY_DIR}/setup.manifest COPYONLY)
    configure_file(${RES}/app.ico ${CMAKE_CURRENT_BINARY_DIR}/app.ico COPYONLY)
    configure_file(${RES}/setup.rc.in ${CMAKE_CURRENT_BINARY_DIR}/setup.rc @ONLY)
    list(APPEND SETUP_SOURCES ${CMAKE_CURRENT_BINARY_DIR}/setup.rc)
endif()

qt_add_executable(AppSetup WIN32 ${SETUP_SOURCES})
target_link_libraries(AppSetup PRIVATE
    Qt6::Widgets Qt6::Svg
    ole32 shell32 uuid advapi32
)
if(SHD_WHITELABEL)
    target_compile_definitions(AppSetup PRIVATE SHD_WHITELABEL)
endif()
set_target_properties(AppSetup PROPERTIES OUTPUT_NAME "AppSetup")
